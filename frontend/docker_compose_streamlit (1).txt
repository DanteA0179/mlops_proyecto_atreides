# Docker Compose para desarrollo local
# Sistema de Optimización Energética - Atreides

version: '3.8'

services:
  # Streamlit UI
  streamlit:
    build:
      context: .
      dockerfile: Dockerfile.streamlit
    container_name: energy-optimizer-ui
    ports:
      - "8501:8501"
    environment:
      - API_URL=http://api:8000
      - OLLAMA_HOST=http://ollama:11434
      - MLFLOW_TRACKING_URI=http://mlflow:5000
    volumes:
      - ./streamlit_app.py:/app/streamlit_app.py
      - ./pages:/app/pages
      - ./.streamlit:/app/.streamlit
    networks:
      - energy-optimizer-network
    depends_on:
      - api
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8501/_stcore/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 5s

  # FastAPI Backend (opcional, si no está corriendo externamente)
  api:
    build:
      context: .
      dockerfile: Dockerfile.api
    container_name: energy-optimizer-api
    ports:
      - "8000:8000"
    environment:
      - MLFLOW_TRACKING_URI=http://mlflow:5000
      - MODEL_PATH=/app/models
    volumes:
      - ./models:/app/models
      - ./data:/app/data
    networks:
      - energy-optimizer-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Ollama (opcional, para Copiloto IA)
  ollama:
    image: ollama/ollama:latest
    container_name: energy-optimizer-ollama
    ports:
      - "11434:11434"
    volumes:
      - ollama-data:/root/.ollama
    networks:
      - energy-optimizer-network
    restart: unless-stopped
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              count: 1
              capabilities: [gpu]

  # MLflow (opcional, para tracking)
  mlflow:
    image: ghcr.io/mlflow/mlflow:v2.9.2
    container_name: energy-optimizer-mlflow
    ports:
      - "5000:5000"
    command: >
      mlflow server
      --host 0.0.0.0
      --port 5000
      --backend-store-uri sqlite:///mlflow/mlflow.db
      --default-artifact-root /mlflow/artifacts
    volumes:
      - mlflow-data:/mlflow
    networks:
      - energy-optimizer-network
    restart: unless-stopped

networks:
  energy-optimizer-network:
    driver: bridge

volumes:
  ollama-data:
  mlflow-data:
